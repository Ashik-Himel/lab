#include <stdio.h>

int main() {
    // Frequency of each element in an array
    int n;
    printf("Enter the size of the array: ");
    scanf("%d", &n);

    int arr1[n];
    printf("Enter all elements of the array: ");
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr1[i]);
    }

    for (int j = 0; j < n; j++) {
        if (arr1[j] != -1) {
            int count = 1;
            for (int k = j+1; k < n; k++) {
                if (arr1[j] == arr1[k]) {
                    count++;
                    arr1[k] = -1;
                }
            }
            printf("%d's frequency in this array is: %d\n", arr1[j], count);
        }
    }
    return 0;
}
